let select = document.querySelector(".select");
let form = document.querySelector(".form");
let polewynik = document.querySelector(".wynik");
let zl = document.querySelector(".zl");

let rateEUR = 4.4278;
let rateGBP = 4.9384;
let rateUSD = 3.9058;

form.addEventListener("submit", (event) => {
    event.preventDefault();

    let amount = +zl.value;
    let currency = select.value;

    let result;

    switch (currency) {
        case "EUR":
            result = amount / rateEUR;
            break;

        case "GBP":
            result = amount / rateGBP;
            break;

        case "USD":
            result = amount / rateUSD;
            break;

    }
    
    polewynik.innerHTML = `${amount.toFixed(2)} PLN = <strong>${result.toFixed(2)} ${currency} </strong>`;
});
